dockerfile
# Use the official MySQL 8.0 image
        FROM mysql:8.0

        # Set root password and create a database
        ENV MYSQL_ROOT_PASSWORD=root
        ENV MYSQL_DATABASE=event
        ENV MYSQL_USER=root

        # Use the native password plugin for better compatibility
        CMD ["--default-authentication-plugin=mysql_native_password"]

        # Expose MySQL port
        EXPOSE 3309

        # Persist data in a volume
        VOLUME /var/lib/mysql

How to use:
    + Build the image:
        docker build -t mysql-event-db .

    + Run the container:
         docker run -d --name mysql-event -p 3309:3306 -e MYSQL_ROOT_PASSWORD=root mysql:8.0

    + Connect to MySQL:
        docker exec -it mysql-event mysql -u root -proot

        OR
             mysql -u root -p
             root
             show databases;
             use database name;
             show tables;
